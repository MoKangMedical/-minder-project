### 2. 核心代码示例

#### Kotlin数据模型
```kotlin
// Thought.kt - 念想数据模型
data class Thought(
    val id: String = UUID.randomUUID().toString(),
    val title: String,                    // 标题
    val description: String = "",         // 描述
    val time: Long,                       // 时间戳
    val location: String? = null,         // 地点
    val category: ThoughtCategory,        // 分类
    val priority: Priority,               // 优先级
    val repeat: RepeatType,               // 重复类型
    val tags: List<String> = emptyList(), // 标签
    val isCompleted: Boolean = false,     // 是否完成
    val completedAt: Long? = null,        // 完成时间
    val createdAt: Long = System.currentTimeMillis(),
    val voiceText: String = "",           // 原始语音文本
    val cardImageUrl: String? = null      // 卡片图片
)

enum class ThoughtCategory {
    FAMILY,    // 亲情 ♥
    WORK,      // 工作 💼
    LIFE,      // 生活 🏃
    STUDY,     // 学习 📚
    OTHER      // 其他
}

enum class Priority {
    HIGH,      // 高 ★★★
    MEDIUM,    // 中 ★★
    LOW        // 低 ★
}

enum class RepeatType {
```

```kotlin
// HomeViewModel.kt
class HomeViewModel(
    private val thoughtRepository: ThoughtRepository,
    private val kimiApi: KimiApi
) : ViewModel() {
    
    private val _thoughts = MutableLiveData<List<Thought>>()
    val thoughts: LiveData<List<Thought>> = _thoughts
    
    private val _greeting = MutableLiveData<String>()
    val greeting: LiveData<String> = _greeting
    
    init {
        loadThoughts()
        updateGreeting()
    }
    
    fun loadThoughts() {
        viewModelScope.launch {
            val allThoughts = thoughtRepository.getAllThoughts()
            _thoughts.value = allThoughts
                .filter { !it.isCompleted }
                .sortedBy { it.time }
        }
    }
    
    fun updateGreeting() {
        val hour = Calendar.getInstance().get(Calendar.HOUR_OF_DAY)
        val userName = "小林" // 从用户设置获取
        
        _greeting.value = when (hour) {
            in 6..10 -> "早安，$userName 👋"
            in 11..17 -> "下午好，$userName ☀️"
            in 18..22 -> "晚上好，$userName 🌙"
            else -> "夜深了，$userName 💤"
        }
    }
    
    fun processVoiceText(voiceText: String) {
        viewModelScope.launch {
            try {
                // 调用Kimi API解析
                val result = kimiApi.parseThought(voiceText)
                
                // 创建念想
                val thought = Thought(
                    title = result.title,
                    description = result.description,
                    time = result.time,
                    location = result.location,
                    category = result.category,
                    priority = result.priority,
                    repeat = result.repeat,
                    tags = result.tags,
                    voiceText = voiceText
                )
                
                // 保存到数据库
                thoughtRepository.insertThought(thought)
                
                // 创建日历事件
                createCalendarEvent(thought)
                
                // 刷新列表
                loadThoughts()
                
            } catch (e: Exception) {
                // 错误处理
                _error.value = "解析失败: ${e.message}"
            }
        }
    }
    
    fun completeThought(thought: Thought) {
        viewModelScope.launch {
            val completedThought = thought.copy(
                isCompleted = true,
                completedAt = System.currentTimeMillis()
            )
            thoughtRepository.updateThought(completedThought)
            
            // 生成完成卡片
            generateCompletionCard(completedThought)
            
            // 显示鼓励
            _encouragement.value = "又完成了一个念想 ♥"
            
            loadThoughts()
        }
    }
}
```
